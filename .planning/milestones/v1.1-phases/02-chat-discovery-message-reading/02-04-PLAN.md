  if (!entity) return null;
  const first = entity.firstName ?? entity.title ?? '';
  const last = entity.lastName ?? '';
  const full = last ? `${first} ${last}` : first;
  return full || null;
}

**Fix 2 (Gap 5): Resolve DM chat names in global search.**

In src/commands/message/search.ts, in the global search branch (around line 77-83), replace the chatTitle extraction:

```typescript
const chatTitle =
  msg.chat?.title || (msg as any)._chat?.title || chatId;
```

with logic that checks for User entities (which have firstName/lastName instead of title):

```typescript
const chat = msg.chat || (msg as any)._chat;
let chatTitle: string;
if (chat?.firstName) {
  // User entity (DM): use firstName + lastName
  const last = chat.lastName ? ` ${chat.lastName}` : '';
  chatTitle = `${chat.firstName}${last}`;
} else {
  chatTitle = chat?.title || chatId;
}
```
